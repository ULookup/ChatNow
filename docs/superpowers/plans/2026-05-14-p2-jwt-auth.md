# P2 JWT 鉴权实施计划（Identity 签发 + Gateway 验签）

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 落地横切 spec §2 定义的生产 JWT 鉴权链路：Identity 服务签发 access/refresh 双 token、滚动刷新 + 重放检测、多 kid 密钥；Gateway 在 HTTP 入口验签并把 user_id/device_id/jti/trace_id 写入 brpc Controller HTTP header metadata（与 P1 的 `extract_auth` 契约对齐）。

**Architecture:** 新增两个 header-only 模块到 `common/auth/`：`jwt_codec.hpp`（HS256 签发/验签 + 多 kid 密钥配置）、`jwt_store.hpp`（Redis 黑名单、refresh 反查、重放链检测）。修订 `proto/identity/identity_service.proto`：删除 `LoginRsp.login_session_id` / `LogoutReq.user_id session_id` / `RefreshTokenReq.session_id` 等老字段，新增 `AuthTokens` 消息及对应字段。`user_server.h` 替换 `_redis_session/_redis_status` 旧逻辑为 `JwtCodec + JwtStore`。`gateway_server.h` 在每个 HTTP 路由头部插入"鉴权中间件"——白名单路由仅生成 trace_id；其余路由解析 `Authorization: Bearer` → 验签 → 写 brpc HTTP metadata。

**Tech Stack:** C++17，[jwt-cpp](https://github.com/Thalhammer/jwt-cpp)（header-only，vendor 到 `third/include/jwt-cpp/jwt.h`），HS256，sw::redis::Redis，jsoncpp（项目已链 `-ljsoncpp`），gtest（项目已用于 file/test）。

**Spec 来源:** `docs/superpowers/specs/2026-05-14-cross-cutting-architecture-design.md` §2.1～§2.5。

- [ ] **Step 1: 写文件全部内容**

```cpp
#pragma once
// JWT 签发 / 验签 — 横切 spec §2.1 §2.2 §2.5
//
// 设计要点：
//   * 仅 HS256（启动时 fail-fast 任何其他 alg）
//   * 多 kid 密钥 map：签发用 current_kid，验签按 token 头部 kid 找密钥
//   * payload: { sub, did, jti, exp, iat, kid, [typ=refresh] }
//   * verify 不查 Redis 黑名单（黑名单查询是 JwtStore 的职责，调用方组合）
//
// 异常：所有失败路径 throw chatnow::ServiceError，code 来自 ErrorCode（1xxx 段）。
//
// 并发：JwtCodec 构造后 _config 不可变；sign/verify 内部线程安全（jwt-cpp 自身线程安全）。

#include <chrono>
#include <cstdint>
#include <random>
#include <stdexcept>
#include <string>
#include <unordered_map>

#include "jwt-cpp/jwt.h"
#include "common/error.pb.h"
#include "common/error/service_error.hpp"

namespace chatnow {

struct JwtConfig {
    std::string current_kid;                                 // e.g. "v1"
    std::unordered_map<std::string, std::string> keys;       // kid -> raw key bytes (>=32 bytes)
    int access_ttl_sec  = 7200;                              // 2h
    int refresh_ttl_sec = 2592000;                           // 30d

    // 启动时调用：current_kid 必须在 keys；每个 key 字节数 >=32；ttl 必须 >0
    // fail-fast：任何不合法 throw std::runtime_error 让 main() 崩
    void validate_or_throw() const;
};

struct JwtClaims {
    std::string sub;       // user_id
    std::string did;       // device_id
    std::string jti;       // 本 token 唯一 id
    std::string kid;       // 签发用 kid
    int64_t     iat_sec = 0;
    int64_t     exp_sec = 0;
    bool        is_refresh = false;
};

class JwtCodec {
public:
    explicit JwtCodec(JwtConfig cfg) : _config(std::move(cfg)) {
        _config.validate_or_throw();
    }

    // 签发 access token；jti 由 caller 提供（也可以传空让 codec 自生成 32 hex）
    std::string sign_access(const std::string& user_id,
                            const std::string& device_id,
                            std::string jti = "");

    // 签发 refresh token（payload 多 typ:"refresh"）
    std::string sign_refresh(const std::string& user_id,
                             const std::string& device_id,
                             std::string jti = "");

    // 验签 + 解析 claims；以下情况抛异常：
    //   非 HS256                       -> ServiceError(AUTH_TOKEN_INVALID)
    //   header.kid 不在 keys           -> ServiceError(AUTH_TOKEN_INVALID)
    //   签名校验失败                   -> ServiceError(AUTH_TOKEN_INVALID)
    //   exp <= now                     -> ServiceError(AUTH_TOKEN_EXPIRED)
    //   require_refresh=true 但 typ!=refresh，或 require_refresh=false 但 typ==refresh
    //                                   -> ServiceError(AUTH_TOKEN_INVALID)
    JwtClaims verify(const std::string& token, bool require_refresh = false);

    int access_ttl_sec()  const { return _config.access_ttl_sec; }
    int refresh_ttl_sec() const { return _config.refresh_ttl_sec; }

    // 工具：32 hex 随机串（供 jti / trace_id 生成；不依赖 OpenSSL 强度）
    static std::string random_hex_32() {
        static thread_local std::mt19937_64 rng{std::random_device{}()};
        static const char* hex = "0123456789abcdef";
        std::string out(32, '0');
        for (int i = 0; i < 32; ++i) out[i] = hex[rng() & 0xF];
        return out;
    }

private:
    JwtConfig _config;

    std::string sign_internal(const std::string& user_id,
                              const std::string& device_id,
                              std::string jti,
                              int ttl_sec,
                              bool is_refresh);
};

}  // namespace chatnow
```

// common/test/test_jwt_codec.cc
#include <gtest/gtest.h>
#include "auth/jwt_codec.hpp"

namespace {
chatnow::JwtConfig make_cfg() {
    chatnow::JwtConfig cfg;
    cfg.current_kid = "v1";
    cfg.keys["v1"] = std::string(32, 'A');   // 32 字节
    cfg.keys["v2"] = std::string(40, 'B');
    cfg.access_ttl_sec  = 60;
    cfg.refresh_ttl_sec = 600;
    return cfg;
}
}  // namespace

TEST(JwtCodec, ValidateRejectsShortKey) {
    chatnow::JwtConfig cfg;
    cfg.current_kid = "v1";
    cfg.keys["v1"]  = std::string(31, 'A');
    EXPECT_THROW(cfg.validate_or_throw(), std::runtime_error);
}

TEST(JwtCodec, ValidateRejectsMissingCurrentKid) {
    chatnow::JwtConfig cfg;
    cfg.current_kid = "v9";
    cfg.keys["v1"]  = std::string(32, 'A');
    EXPECT_THROW(cfg.validate_or_throw(), std::runtime_error);
}

TEST(JwtCodec, AccessRoundtrip) {
    chatnow::JwtCodec codec(make_cfg());
    auto tok = codec.sign_access("u_1", "d_1", "jti_x");
    auto claims = codec.verify(tok, /*require_refresh=*/false);
    EXPECT_EQ(claims.sub, "u_1");
    EXPECT_EQ(claims.did, "d_1");
    EXPECT_EQ(claims.jti, "jti_x");
    EXPECT_EQ(claims.kid, "v1");
    EXPECT_FALSE(claims.is_refresh);
    EXPECT_GT(claims.exp_sec, claims.iat_sec);
}

TEST(JwtCodec, RefreshRoundtrip) {
    chatnow::JwtCodec codec(make_cfg());
    auto tok = codec.sign_refresh("u_1", "d_1");
    auto claims = codec.verify(tok, /*require_refresh=*/true);
    EXPECT_TRUE(claims.is_refresh);
    EXPECT_EQ(claims.sub, "u_1");
    EXPECT_FALSE(claims.jti.empty());
}

TEST(JwtCodec, VerifyRejectsAccessWhenRefreshRequired) {
    chatnow::JwtCodec codec(make_cfg());
    auto tok = codec.sign_access("u", "d");
    EXPECT_THROW(codec.verify(tok, /*require_refresh=*/true), chatnow::ServiceError);
}

TEST(JwtCodec, VerifyRejectsRefreshWhenAccessExpected) {
    chatnow::JwtCodec codec(make_cfg());
    auto tok = codec.sign_refresh("u", "d");

inline void JwtConfig::validate_or_throw() const {
    if (current_kid.empty()) {
        throw std::runtime_error("auth.jwt.current_kid is empty");
    }
    if (keys.find(current_kid) == keys.end()) {
        throw std::runtime_error("auth.jwt.current_kid not in keys: " + current_kid);
    }
    if (access_ttl_sec <= 0) {
        throw std::runtime_error("auth.jwt.access_ttl_sec must be > 0");
    }
    if (refresh_ttl_sec <= 0) {
        throw std::runtime_error("auth.jwt.refresh_ttl_sec must be > 0");
    }
    for (const auto& kv : keys) {
        if (kv.second.size() < 32) {
            throw std::runtime_error("auth.jwt.keys[" + kv.first +
                                     "] must be >=32 bytes, got " +
                                     std::to_string(kv.second.size()));
        }
    }

inline std::string JwtCodec::sign_internal(const std::string& user_id,
                                           const std::string& device_id,
                                           std::string jti,
                                           int ttl_sec,
                                           bool is_refresh) {
    if (jti.empty()) jti = random_hex_32();
    const auto& kid = _config.current_kid;
    const auto& key = _config.keys.at(kid);
    auto now = std::chrono::system_clock::now();

    auto builder = jwt::create()
        .set_type("JWT")
        .set_key_id(kid)
        .set_issued_at(now)
        .set_expires_at(now + std::chrono::seconds(ttl_sec))
        .set_payload_claim("sub", jwt::claim(user_id))
        .set_payload_claim("did", jwt::claim(device_id))
        .set_payload_claim("jti", jwt::claim(jti));
    if (is_refresh) {
        builder = builder.set_payload_claim("typ", jwt::claim(std::string("refresh")));
    }
    return builder.sign(jwt::algorithm::hs256{key});
}

inline std::string JwtCodec::sign_access(const std::string& user_id,
                                         const std::string& device_id,
                                         std::string jti) {
    return sign_internal(user_id, device_id, std::move(jti),
                         _config.access_ttl_sec, /*is_refresh=*/false);
}

inline std::string JwtCodec::sign_refresh(const std::string& user_id,
                                          const std::string& device_id,
                                          std::string jti) {
    return sign_internal(user_id, device_id, std::move(jti),
                         _config.refresh_ttl_sec, /*is_refresh=*/true);
}

inline JwtClaims JwtCodec::verify(const std::string& token, bool require_refresh) {
    using chatnow::common::ErrorCode;

    jwt::decoded_jwt<jwt::traits::kazuho_picojson> decoded;
    try {
        decoded = jwt::decode(token);
    } catch (const std::exception&) {
        throw ServiceError(ErrorCode::AUTH_TOKEN_INVALID, "token decode failed");
    }
    if (decoded.get_algorithm() != "HS256") {
        throw ServiceError(ErrorCode::AUTH_TOKEN_INVALID, "alg not HS256");
    }
    if (!decoded.has_key_id()) {
        throw ServiceError(ErrorCode::AUTH_TOKEN_INVALID, "missing kid");
    }
    const auto kid = decoded.get_key_id();
    auto it = _config.keys.find(kid);
    if (it == _config.keys.end()) {
        throw ServiceError(ErrorCode::AUTH_TOKEN_INVALID, "unknown kid: " + kid);
    }

    try {
        auto verifier = jwt::verify()
            .allow_algorithm(jwt::algorithm::hs256{it->second})
            .leeway(0);
        verifier.verify(decoded);
    } catch (const jwt::error::token_verification_exception& e) {
        // jwt-cpp 把过期也归到 token_verification_exception；区分错误信息
        std::string what = e.what();
        if (what.find("expired") != std::string::npos) {
            throw ServiceError(ErrorCode::AUTH_TOKEN_EXPIRED, "token expired");
        }
        throw ServiceError(ErrorCode::AUTH_TOKEN_INVALID, "verify failed: " + what);
    } catch (const std::exception& e) {
        throw ServiceError(ErrorCode::AUTH_TOKEN_INVALID,
                           std::string("verify failed: ") + e.what());
    }

    JwtClaims out;
    out.kid = kid;
    try {
        out.sub = decoded.get_payload_claim("sub").as_string();
        out.did = decoded.get_payload_claim("did").as_string();
        out.jti = decoded.get_payload_claim("jti").as_string();
    } catch (const std::exception&) {
        throw ServiceError(ErrorCode::AUTH_TOKEN_INVALID, "claims missing sub/did/jti");
    }
    out.iat_sec = std::chrono::duration_cast<std::chrono::seconds>(
        decoded.get_issued_at().time_since_epoch()).count();
    out.exp_sec = std::chrono::duration_cast<std::chrono::seconds>(
        decoded.get_expires_at().time_since_epoch()).count();

    bool has_typ_refresh = false;
    if (decoded.has_payload_claim("typ")) {
        try {
            has_typ_refresh = (decoded.get_payload_claim("typ").as_string() == "refresh");
        } catch (...) { has_typ_refresh = false; }
    }
    out.is_refresh = has_typ_refresh;
    if (require_refresh != has_typ_refresh) {
        throw ServiceError(ErrorCode::AUTH_TOKEN_INVALID,
                           require_refresh ? "expected refresh token"
                                           : "unexpected refresh token");
    }
    return out;
}

#pragma once
// JWT Redis 三类 key 操作 — 横切 spec §2.3
//
//   im:jwt:revoked:{jti}                  -> "1"     TTL = 剩余寿命
//   im:jwt:rt:{user_id}:{device_id}       -> refresh_jti
//   im:jwt:rt_chain:{old_jti}             -> "rotated"  TTL = 24h
//
// 失败模式：所有底层 redis 抛异常时，函数自身吞掉异常 + LOG_ERROR + 返回保守值
// （revoke 检查失败 → 视为未撤销，避免雪崩；写失败 → 仅日志，调用方按业务决定）
//
// 重放检测语义：rotate_refresh_or_detect_reuse 是核心状态机，详见函数注释。

#include <chrono>
#include <memory>
#include <string>

#include <sw/redis++/redis++.h>

#include "infra/logger.hpp"

namespace chatnow {

namespace jwt_key {
inline constexpr const char* kRevoked  = "im:jwt:revoked:";
inline constexpr const char* kRefresh  = "im:jwt:rt:";
inline constexpr const char* kRtChain  = "im:jwt:rt_chain:";
}

class JwtStore {
public:
    using ptr = std::shared_ptr<JwtStore>;
    explicit JwtStore(std::shared_ptr<sw::redis::Redis> c) : _c(std::move(c)) {}

    // 黑名单：写 revoked:{jti}，TTL = ttl_sec（应等于 token 剩余寿命）
    void revoke(const std::string& jti, int ttl_sec);

    // 查 revoked:{jti} 是否命中
    bool is_revoked(const std::string& jti);

    // 写 rt:{uid}:{did} = refresh_jti，TTL = ttl_sec（refresh token 寿命）
    void put_active_refresh(const std::string& user_id,
                            const std::string& device_id,
                            const std::string& refresh_jti,
                            int ttl_sec);

    // 读 rt:{uid}:{did}；空表示无活跃 refresh（未登录 / 已 logout）
    std::string get_active_refresh(const std::string& user_id,
                                   const std::string& device_id);

    // 删 rt:{uid}:{did}（logout / kick 时用）
    void clear_active_refresh(const std::string& user_id,
                              const std::string& device_id);

    enum class RotateResult { kOk, kReuseDetected };

    // 滚动刷新核心：原子地完成
    //   1. 检查 rt_chain:{old_jti} 是否存在 → 存在则返回 kReuseDetected
    //   2. 写 rt_chain:{old_jti} = "rotated" TTL=24h
    //   3. 更新 rt:{uid}:{did} = new_refresh_jti，TTL=new_refresh_ttl_sec
    //
    // 注意：1+2 用 SET key val NX EX 保证原子；步骤 3 是单独 SET，
    //      若 3 失败仅记录，调用方仍可用步骤 1 的成功语义颁发新 token。
    RotateResult rotate_refresh_or_detect_reuse(const std::string& user_id,
                                                const std::string& device_id,
                                                const std::string& old_refresh_jti,
                                                const std::string& new_refresh_jti,
                                                int new_refresh_ttl_sec);

private:
    std::shared_ptr<sw::redis::Redis> _c;

    static constexpr int kChainTtlSec = 24 * 3600;
};

}  // namespace chatnow

inline void JwtStore::revoke(const std::string& jti, int ttl_sec) {
    if (jti.empty() || ttl_sec <= 0) return;
    try {
        _c->set(jwt_key::kRevoked + jti, "1", std::chrono::seconds(ttl_sec));
    } catch (const std::exception& e) {
        LOG_ERROR("JwtStore.revoke 失败 jti={}: {}", jti, e.what());
    }
}

inline bool JwtStore::is_revoked(const std::string& jti) {
    if (jti.empty()) return false;
    try {
        auto v = _c->get(jwt_key::kRevoked + jti);
        return v.has_value();
    } catch (const std::exception& e) {
        LOG_ERROR("JwtStore.is_revoked 失败 jti={}: {}", jti, e.what());
        return false;  // 保守：不阻断业务
    }
}

inline void JwtStore::put_active_refresh(const std::string& user_id,
                                         const std::string& device_id,
                                         const std::string& refresh_jti,
                                         int ttl_sec) {
    if (user_id.empty() || device_id.empty() || refresh_jti.empty()) return;
    try {
        _c->set(jwt_key::kRefresh + user_id + ":" + device_id,
                refresh_jti,
                std::chrono::seconds(ttl_sec));
    } catch (const std::exception& e) {
        LOG_ERROR("JwtStore.put_active_refresh 失败 u={} d={}: {}",
                  user_id, device_id, e.what());
    }
}

                                                const std::string& device_id) {
    try {
        auto v = _c->get(jwt_key::kRefresh + user_id + ":" + device_id);
        return v.value_or("");
    } catch (const std::exception& e) {
        LOG_ERROR("JwtStore.get_active_refresh 失败 u={} d={}: {}",
                  user_id, device_id, e.what());
        return "";
    }
}

inline void JwtStore::clear_active_refresh(const std::string& user_id,
                                           const std::string& device_id) {
    try {
        _c->del(jwt_key::kRefresh + user_id + ":" + device_id);
    } catch (const std::exception& e) {
        LOG_ERROR("JwtStore.clear_active_refresh 失败 u={} d={}: {}",
                  user_id, device_id, e.what());

}

inline JwtStore::RotateResult JwtStore::rotate_refresh_or_detect_reuse(
        const std::string& user_id,
        const std::string& device_id,
        const std::string& old_refresh_jti,
        const std::string& new_refresh_jti,
        int new_refresh_ttl_sec) {
    try {
        // SET rt_chain:{old_jti} "rotated" NX EX 24h
        // 命中 NX 失败 → 已 rotate 过 → 这是攻击者第二次用旧 refresh
        auto ok = _c->set(jwt_key::kRtChain + old_refresh_jti,
                          "rotated",
                          std::chrono::seconds(kChainTtlSec),
                          sw::redis::UpdateType::NOT_EXIST);
        if (!ok) {
            return RotateResult::kReuseDetected;
        }
    } catch (const std::exception& e) {
        LOG_ERROR("JwtStore.rotate chain SET 失败 old_jti={}: {}", old_refresh_jti, e.what());
        // 链路写失败：保守按"未被重放"放行；下次会再尝试
    }
    put_active_refresh(user_id, device_id, new_refresh_jti, new_refresh_ttl_sec);
    return RotateResult::kOk;
}

#pragma once
// 从 JSON 文件加载 JwtConfig — 横切 spec §2.2
// 启动时 fail-fast：文件不存在 / JSON 解析失败 / 校验失败 → throw std::runtime_error
//
// JSON shape（spec §2.2）:
// {
//   "auth": {
//     "jwt": {
//       "current_kid": "v1",
//       "keys": { "v1": "<base64 string ≥32 raw bytes>" },
//       "access_ttl_sec":  7200,
//       "refresh_ttl_sec": 2592000
//     }
//   }
// }
//
// 注意：keys 的 value 视作 raw bytes 直接喂给 HS256（不做 base64 解码），
// 因此配置文件里写的字符串本身的字节数 >= 32 即可（可全部为 ASCII）。

#include <fstream>
#include <sstream>
#include <stdexcept>
#include <string>

#include <json/json.h>   // jsoncpp（项目已链接 -ljsoncpp）

#include "auth/jwt_codec.hpp"

namespace chatnow {

inline JwtConfig load_jwt_config_from_file(const std::string& path) {
    std::ifstream ifs(path);
    if (!ifs) {
        throw std::runtime_error("auth_config: cannot open " + path);
    }
    Json::Value root;
    Json::CharReaderBuilder b;
    std::string err;
    if (!Json::parseFromStream(b, ifs, &root, &err)) {
        throw std::runtime_error("auth_config: parse failed: " + err);
    }
    if (!root.isMember("auth") || !root["auth"].isMember("jwt")) {
        throw std::runtime_error("auth_config: missing auth.jwt section in " + path);
    }
    const auto& j = root["auth"]["jwt"];

    JwtConfig cfg;
    cfg.current_kid = j.get("current_kid", "").asString();
    if (j.isMember("access_ttl_sec"))  cfg.access_ttl_sec  = j["access_ttl_sec"].asInt();
    if (j.isMember("refresh_ttl_sec")) cfg.refresh_ttl_sec = j["refresh_ttl_sec"].asInt();
    if (!j.isMember("keys") || !j["keys"].isObject()) {
        throw std::runtime_error("auth_config: missing auth.jwt.keys map");
    }
    for (const auto& kid : j["keys"].getMemberNames()) {

                       const ::chatnow::identity::LoginReq* request,
                       ::chatnow::identity::LoginRsp* response,
                       ::google::protobuf::Closure* done)
    {
        brpc::ClosureGuard rpc_guard(done);
        auto* header = response->mutable_header();
        header->set_request_id(request->request_id());
        try {
            // 1. 凭据校验（仅支持 username_pwd；phone_code 在 P3+ 处理）

                                   "phone_code login not supported in P2");

            auto user = _mysql_user->select_by_nickname(cred.username());
            if (!user || user->password() != cred.password()) {
                throw ServiceError(common::ErrorCode::AUTH_INVALID_CREDENTIALS,
                                   "username or password incorrect");

            // 2. device_id 取自请求（P2 不强制校验，P3 收紧）
            std::string device_id = request->device_id();

     * 需要 metadata：x-user-id / x-device-id / x-jwt-jti（P1 extract_auth 校验）

    virtual void Logout(::google::protobuf::RpcController* controller,
                        const ::chatnow::identity::LogoutReq* request,
                        ::chatnow::identity::LogoutRsp* response,
                        ::google::protobuf::Closure* done)
    {
        brpc::ClosureGuard rpc_guard(done);
        auto* cntl = static_cast<brpc::Controller*>(controller);
        auto* header = response->mutable_header();

### Task 13: 实现 IdentityService.RefreshToken RPC（重放检测 + 滚动刷新）

**Files:**
- Modify: `user/source/user_server.h`

- [ ] **Step 1: 在 Logout 实现下方追加 RefreshToken handler**

```cpp
    /* brief: IdentityService.RefreshToken — 滚动刷新 + 重放检测（横切 spec §2.3）
     *  入参：refresh_token（不依赖入站 metadata，因为客户端可能 access 已过期）
     */
    virtual void RefreshToken(::google::protobuf::RpcController* controller,
                              const ::chatnow::identity::RefreshTokenReq* request,
                              ::chatnow::identity::RefreshTokenRsp* response,
                              ::google::protobuf::Closure* done)
    {
        brpc::ClosureGuard rpc_guard(done);
        auto* header = response->mutable_header();
        header->set_request_id(request->request_id());
        try {
            // 1. 验签 refresh token（强制 typ=refresh）
            auto claims = _jwt_codec->verify(request->refresh_token(), /*require_refresh=*/true);
            // 2. 黑名单检查
            if (_jwt_store->is_revoked(claims.jti)) {
                throw ServiceError(common::ErrorCode::AUTH_TOKEN_INVALID,
                                   "refresh token revoked");
            }
            // 3. 反查表是否还指向这个 refresh_jti
            std::string active = _jwt_store->get_active_refresh(claims.sub, claims.did);
            if (active != claims.jti) {
                // 不一致 = 该 refresh 已被替换（可能有人用旧 refresh）→ 视为重放
                // 同步进入 reuse 路径：吊销该设备所有 token + 清反查表
                _jwt_store->revoke(claims.jti, _jwt_codec->refresh_ttl_sec());
                _jwt_store->clear_active_refresh(claims.sub, claims.did);
                throw ServiceError(common::ErrorCode::AUTH_REFRESH_TOKEN_REUSED,
                                   "refresh token mismatch active");
            }
            // 4. 颁发新 access + 新 refresh
            std::string new_access_jti  = JwtCodec::random_hex_32();
            std::string new_refresh_jti = JwtCodec::random_hex_32();
            std::string new_access  = _jwt_codec->sign_access(claims.sub, claims.did, new_access_jti);
            std::string new_refresh = _jwt_codec->sign_refresh(claims.sub, claims.did, new_refresh_jti);
            // 5. 原子滚动 + 重放链检测
            auto rot = _jwt_store->rotate_refresh_or_detect_reuse(
                claims.sub, claims.did, claims.jti, new_refresh_jti,
                _jwt_codec->refresh_ttl_sec());
            if (rot == JwtStore::RotateResult::kReuseDetected) {
                _jwt_store->revoke(claims.jti, _jwt_codec->refresh_ttl_sec());
                _jwt_store->clear_active_refresh(claims.sub, claims.did);
                throw ServiceError(common::ErrorCode::AUTH_REFRESH_TOKEN_REUSED,
                                   "refresh chain reuse detected");
            }
            // 6. 旧 refresh 写黑名单（剩余寿命）
            int64_t now = std::chrono::duration_cast<std::chrono::seconds>(
                std::chrono::system_clock::now().time_since_epoch()).count();
            int remain = static_cast<int>(claims.exp_sec - now);
            if (remain > 0) {
                _jwt_store->revoke(claims.jti, remain);
            }

            auto* tokens = response->mutable_tokens();
            tokens->set_access_token(new_access);
            tokens->set_refresh_token(new_refresh);
            tokens->set_access_expires_in_sec(_jwt_codec->access_ttl_sec());
            tokens->set_refresh_expires_in_sec(_jwt_codec->refresh_ttl_sec());

            header->set_success(true);
            header->set_error_code(common::ErrorCode::OK);
        } catch (const ServiceError& e) {
            header->set_success(false);
            header->set_error_code(e.code());
            header->set_error_message(e.message());
            LOG_WARN("RefreshToken 失败 rid={} code={}",
                     request->request_id(), e.code());
        } catch (const std::exception& e) {
            header->set_success(false);
            header->set_error_code(common::ErrorCode::SYSTEM_INTERNAL_ERROR);
            header->set_error_message("internal error");
            LOG_ERROR("RefreshToken 异常 rid={}: {}", request->request_id(), e.what());
        }
    }

        return JwtCodec::random_hex_32();
    }

    /* brief: AuthInfo — middleware 输出 */
    struct AuthInfo {
        bool        authed = false;     // 鉴权路径走过且通过
        std::string user_id;
        std::string device_id;
        std::string jwt_jti;
        std::string trace_id;           // 一定有
    };

    /* brief: 写入 brpc Controller 的 HTTP 请求 metadata（与 P1 extract_auth 契约对齐） */
    void _apply_auth_to_brpc(brpc::Controller &cntl, const AuthInfo &a) {
        cntl.http_request().SetHeader(kMetaTraceId, a.trace_id);
        if (a.authed) {
            cntl.http_request().SetHeader(kMetaUserId,   a.user_id);
            cntl.http_request().SetHeader(kMetaDeviceId, a.device_id);
            cntl.http_request().SetHeader(kMetaJwtJti,   a.jwt_jti);
        }
    }

    /* brief: HTTP 入口鉴权中间件（白名单仅生成 trace_id；其余路径必须 JWT 验签）
     *
     * 返回：
     *   AuthInfo{authed=true,...}   验签通过；调用方应 _apply_auth_to_brpc 后转发
     *   AuthInfo{authed=false,...}  白名单路径（仅生成 trace_id）
     *   抛 ServiceError 表示鉴权失败 — 调用方按错误码写 401-style 响应
     */
    AuthInfo _jwt_middleware(const httplib::Request &request, bool whitelisted) {
        AuthInfo a;
        // trace_id：客户端送 X-Trace-Id 且为 32 hex 则透传，否则生成
        auto incoming = request.get_header_value("X-Trace-Id");
        if (incoming.size() == 32 &&
            incoming.find_first_not_of("0123456789abcdefABCDEF") == std::string::npos) {
            a.trace_id = incoming;
        } else {
            a.trace_id = _gen_trace_id();
        }
        if (whitelisted) return a;

        auto auth_header = request.get_header_value("Authorization");
        const std::string kPrefix = "Bearer ";
        if (auth_header.size() <= kPrefix.size() ||
            auth_header.compare(0, kPrefix.size(), kPrefix) != 0) {
            throw ServiceError(common::ErrorCode::AUTH_TOKEN_INVALID,
                               "missing Authorization Bearer");
        }
        std::string token = auth_header.substr(kPrefix.size());
        auto claims = _jwt_codec->verify(token, /*require_refresh=*/false);
        if (_jwt_store->is_revoked(claims.jti)) {
            throw ServiceError(common::ErrorCode::AUTH_TOKEN_INVALID, "token revoked");
        }
        a.authed    = true;
        a.user_id   = claims.sub;
        a.device_id = claims.did;
        a.jwt_jti   = claims.jti;
        return a;
    }

    /* brief: 把 ServiceError 写成 ResponseHeader-only 的统一错误回包 */
    template <typename Rsp>
    void _write_auth_error(httplib::Response &response, const ServiceError &e,
                           const std::string &request_id, const std::string &trace_id) {
        Rsp rsp;
        auto* h = rsp.mutable_header();
        h->set_request_id(request_id);
        h->set_success(false);
        h->set_error_code(e.code());
        h->set_error_message(e.message());
        response.set_header("X-Trace-Id", trace_id);
        response.status = (e.code() == common::ErrorCode::AUTH_TOKEN_EXPIRED ||
                           e.code() == common::ErrorCode::AUTH_TOKEN_INVALID) ? 401 : 200;
        response.set_content(rsp.SerializeAsString(), "application/x-protobuf");
    }
```

Standard Response Envelope & Error Codes

Every response shares the same structure; every error has a code client can switch on

Problem Today

Inconsistent: Some Rsp have `success + errmsg`, some don't. Error is always a string — client can't branch on it.
No pagination standard: GetHistory uses `start_time/over_time`, GetRecent uses `msg_count`, GetOffline uses `last_message_id + msg_count` — three different patterns for essentially "give me N items after cursor".
common/envelope.proto
syntax = "proto3";
package chatnow.common;

// Every response embeds this as field 1
message ResponseHeader {
  string request_id = 1;   // echo back
  bool success = 2;        // false → check error_code
  int32 error_code = 3;    // 0 = OK, see error.proto
  string error_message = 4;// human-readable, debug only
}

// Forward pagination: "give me N items after cursor"
message PageRequest {
  int32 limit = 1;         // max 100, default 20
  string cursor = 2;       // opaque cursor from last response
}

message PageResponse {
  bool has_more = 1;       // true → more items available
  string next_cursor = 2;  // pass to next PageRequest
  int32 total_count = 3;   // optional, may be approximate
}

// Time-range pagination (legacy / analytics)
message TimeRange {
  int64 start_time_ms = 1;
  int64 end_time_ms = 2;
}
common/error.proto
enum ErrorCode {
  OK = 0;

  // 1000-1999: Auth / Identity
  AUTH_INVALID_CREDENTIALS = 1001;
  AUTH_TOKEN_EXPIRED = 1002;
  AUTH_TOKEN_INVALID = 1003;
  AUTH_USER_NOT_FOUND = 1004;
  AUTH_USER_ALREADY_EXISTS = 1005;
  AUTH_VERIFY_CODE_INVALID = 1006;
  AUTH_VERIFY_CODE_EXPIRED = 1007;

  // 2000-2999: Relationship
  RELATIONSHIP_ALREADY_FRIENDS = 2001;
  RELATIONSHIP_NOT_FRIENDS = 2002;
  RELATIONSHIP_BLOCKED = 2003;
  RELATIONSHIP_REQUEST_PENDING = 2004;

  // 3000-3999: Conversation
  CONVERSATION_NOT_FOUND = 3001;
  CONVERSATION_NOT_MEMBER = 3002;
  CONVERSATION_NO_PERMISSION = 3003;
  CONVERSATION_MEMBER_LIMIT = 3004;

  MESSAGE_NOT_FOUND = 4001;

  // 5000-5999: Media

}

Usage Example

// BEFORE (current pattern — every Rsp is different)
message GetFriendListRsp {
  string request_id = 1;
  bool success = 2;
  string errmsg = 3;                     // "Friend not found" — string, unswitchable
  repeated UserInfo friend_list = 4;     // no pagination — all at once
}

// AFTER (standard pattern)
message GetFriendListRsp {
  ResponseHeader header = 1;             // always field 1
  PageResponse page = 2;                 // always field 2 when paginated
  repeated FriendItem items = 3;         // domain data starts at field 3+
}

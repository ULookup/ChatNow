Presence 域：在线状态 + 输入中指示 + 最后在线时间

全新服务。当前在线状态散落在 Push（im:online:{uid} SET），没有面向客户端的状态 API

Presence 状态机

enum PresenceState {
  PRESENCE_UNSPECIFIED = 0;
  ONLINE = 1;       // 在线（心跳活跃）
  AWAY = 2;         // 离开（心跳超时，尚未离线）
  BUSY = 3;         // 请勿打扰
  OFFLINE = 4;      // 离线
  INVISIBLE = 5;    // 隐身（对他人显示离线，自己正常收发）
}

message Presence {
  string user_id = 1;
  PresenceState state = 2;
  int64 last_active_at_ms = 3;         // 最后活跃时间
  repeated string active_device_types = 4; // ["mobile","desktop"]
  optional string custom_status = 5;  // 自定义状态文字
}
presence/presence_service.proto
service PresenceService {
  // === 状态管理 ===
  rpc SetPresence(SetPresenceReq)
    returns (SetPresenceRsp);
  rpc GetPresence(GetPresenceReq)
    returns (GetPresenceRsp);
  rpc BatchGetPresence(BatchGetPresenceReq)
    returns (BatchGetPresenceRsp);     // 好友列表批量查询

  // === 订阅推送 ===
  // 客户端订阅好友上线/下线变化
  rpc SubscribePresence(SubscribeReq)
    returns (SubscribeRsp);
  rpc UnsubscribePresence(UnsubscribeReq)
    returns (UnsubscribeRsp);

  // === 输入中指示 ===
  rpc SendTypingIndicator(TypingReq)
    returns (TypingRsp);

  // === 内部接口 ===
  // Push 服务心跳续期时调用
  rpc Heartbeat(HeartbeatReq)
    returns (HeartbeatRsp);
}
Typing 和 Heartbeat 结构
// 输入中指示（客户端→Push→Presence）
message TypingReq {
  string request_id = 1;
  string user_id = 2;
  string conversation_id = 3;
  bool is_typing = 4;  // true=开始输入, false=停止
}
// 服务端收到后推送给会话其他成员

// 心跳（Push→Presence，内部接口）
message HeartbeatReq {
  string request_id = 1;
  string user_id = 2;
  string device_id = 3;
  string device_type = 4;    // mobile/desktop/web
  bool is_connected = 5;     // ws connected
}

与其他服务的交互

客户端                     Push 服务                  Presence 服务
  │                          │                           │
  │── WS CLIENT_HEARTBEAT ──>│                           │
  │                          │── Heartbeat RPC ─────────>│  (每 60s)
  │                          │                           │  redis SADD im:online:{uid} device_id
  │                          │                           │  ZADD im:lastseen:{uid} timestamp
  │                          │                           │
  │── SetPresence RPC ─────────────────────────────────>│  (用户切 BUSY/INVISIBLE)
  │                          │                           │
  │── TypingIndicator ──────>│── Typing RPC ────────────>│
  │                          │                           │  → 推送给会话其他在线成员
  │<── PresenceChange NOTIFY ─│<── Subscribe 事件 ──────│

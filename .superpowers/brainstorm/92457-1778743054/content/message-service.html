Message Service + 内部消息 + 完整拓扑

message/ 域三个文件：message_types.proto + message_service.proto + message_internal.proto

message/message_service.proto — 客户端 RPC

service MessageService {
  // === 核心同步 ===
  rpc SyncMessages(SyncMessagesReq) returns (SyncMessagesRsp);     // 增量同步（seq 游标）
  rpc GetHistory(GetHistoryReq) returns (GetHistoryRsp);           // 时间段查询

  // === 批量/单条 ===
  rpc GetMessagesById(GetMessagesByIdReq) returns (GetMessagesByIdRsp);

  // === 搜索 ===
  rpc SearchMessages(SearchMessagesReq) returns (SearchMessagesRsp);

  // === 撤回 ===                                              ← 全新
  rpc RecallMessage(RecallMessageReq) returns (RecallMessageRsp);

  // === Reactions ===                                         ← 全新
  rpc AddReaction(AddReactionReq) returns (AddReactionRsp);
  rpc RemoveReaction(RemoveReactionReq) returns (RemoveReactionRsp);
  rpc GetReactions(GetReactionsReq) returns (GetReactionsRsp);

  // === Pin ===                                               ← 全新
  rpc PinMessage(PinMessageReq) returns (PinMessageRsp);
  rpc UnpinMessage(UnpinMessageReq) returns (UnpinMessageRsp);
  rpc ListPinnedMessages(ListPinnedReq) returns (ListPinnedRsp);

  // === 管理 ===
  rpc DeleteMessages(DeleteMessagesReq) returns (DeleteMessagesRsp);   // 从 timeline 删除
  rpc ClearConversation(ClearConversationReq) returns (ClearConversationRsp); // 清空

  // === 内部接口 ===
  rpc SelectByClientMsgId(SelectByClientMsgIdReq) returns (SelectByClientMsgIdRsp); // 幂等去重
  rpc UpdateReadAck(UpdateReadAckReq) returns (UpdateReadAckRsp);       // Push 上报 ACK
}
关键 RPC 说明
// 增量同步 — 替代 GetOfflineMsg
message SyncMessagesReq {
  string request_id = 1;
  string conversation_id = 2;
  uint64 after_seq = 3;     // 客户端本地最大 seq
  int32 limit = 4;          // 默认 50，最大 200
}
message SyncMessagesRsp {
  ResponseHeader header = 1;
  repeated Message messages = 2;
  bool has_more = 3;
  uint64 latest_seq = 4;    // 服务端最新 seq
}

// 撤回 — 2 分钟内有效
message RecallMessageReq {
  string request_id = 1;
  string conversation_id = 2;
  int64 message_id = 3;
}
// 服务端执行：UPDATE status=RECALLED
//           → 推送 NotifyMessageRecalled 给全体成员
message/message_internal.proto
// MQ 投递的胖消息（DB 落库用）
message InternalMessage {
  Message message = 1;
  repeated string member_id_list = 2;
  repeated UserSeqPair user_seqs = 3;  // uid→seq
  bool is_large_group = 4;             // >200 人走读扩散
}

// ES 索引事件（DB commit 后投递）
message ESIndexEvent {
  int64 message_id = 1;
  string conversation_id = 2;
  string sender_id = 3;
  string content_text = 4;            // 纯文本内容
  int64 created_at_ms = 5;
  uint64 seq_id = 6;
  MessageType message_type = 7;
}

// 透传给 Push 的通知包装
message PushNotify {
  InternalMessage internal_msg = 1;
  string push_to_instance = 2;        // 目标 Push 实例
}
